## Quick Example

```typescript
module hello {
	fn greet(name string) {
		match validate_name(name) {
			ok(valid_name) => console.log("Hello,", valid_name),
			error(msg) => console.error("Error:", msg)
		}
	}

	fn validate_name(name string) {
		if name.length() == 0 {
			return error("Name cannot be empty");
		}
		return ok(name);
	}

	fn main() {
		greet("R2D2");  // Prints: Hello, R2D2
		greet("");      // Prints: Error: Name cannot be empty
	}
}
```
